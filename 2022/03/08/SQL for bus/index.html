建表

地理位置表

create table t_geo  select id, gps_code, gps_direction from t_bus_status;

路线表

create table t_road select distinct road_id, road_name, site_id, site_no, site_name from t_bus_status;

展示在Grafana

线路的车次

SELECT
  $__timeGroupAlias(gps_date,$__interval),
  road_id AS metric,
  count(site_id) AS "site_id"
FROM t_bus_status
WHERE
  $__timeFilter(gps_date)
GROUP BY 1,2
ORDER BY $__timeGroup(gps_date,$__interval)
