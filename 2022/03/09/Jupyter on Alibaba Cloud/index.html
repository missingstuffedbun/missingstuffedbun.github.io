bus_id varchar(255) DEFAULT NULL,
bus_plate varchar(255) DEFAULT NULL,
bus_state int(11) DEFAULT NULL,
comp_code varchar(255) DEFAULT NULL,
current_runs int(11) DEFAULT NULL,
gps_code varchar(255) DEFAULT NULL,
gps_date varchar(255) DEFAULT NULL,
store_date varchar(255) DEFAULT NULL,
gps_direction int(11) DEFAULT NULL,
gps_mileage varchar(255) DEFAULT NULL,
gps_speed varchar(255) DEFAULT NULL,
plan_runs int(11) DEFAULT NULL,
plan_time varchar(255) DEFAULT NULL,
road_id varchar(255) DEFAULT NULL,
road_name varchar(255) DEFAULT NULL,
road_type int(11) DEFAULT NULL,
site_id varchar(255) DEFAULT NULL,
site_name varchar(255) DEFAULT NULL,
site_no int(11) DEFAULT NULL,
site_state int(11) DEFAULT NULL,
create_time varchar(255) DEFAULT NULL,
PRIMARY KEY (id)
) ENGINE=InnoDB AUTO_INCREMENT=24771706 DEFAULT CHARSET=utf8;
